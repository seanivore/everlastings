Placeholder Hygiene

Track B's HTML pages contain hardcoded data that Track C will replace with live Supabase fetches. Every block of hardcoded content must be wrapped in a PLACEHOLDER: comment so Track C can grep them. One wrapper per logical block, not per atom.

Use atom-level wrapping ONLY when fields will be replaced independently (e.g., hero image vs. hero copy that come from different data sources).

<!-- PLACEHOLDER: product-sticky-card -->
<aside class="product-sticky-card">
  <h2>The Sunkeeper</h2>
  <p class="price">$245.00</p>
</aside>
<!-- /PLACEHOLDER -->

/* PLACEHOLDER: hero-bg */

/* /PLACEHOLDER */

// PLACEHOLDER: sample-cart-data

CustomEvent Contracts (for Track C)

'email-cta-submit'

window.addEventListener('email-cta-submit', (e) => {
  // e.detail = {
  //   source: 'product-interest' | 'cart-exit' | 'contemplation-offer'
  //         | 'newsletter-footer' | 'newsletter-shop-empty',
  //   email: string,
  //   productSlug?: string,   // present for 'product-interest' only
  // }
  fetch('/api/subscribe', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(e.detail),
  });
});

'consent-change'

window.addEventListener('consent-change', (e) => {
  // e.detail = {
  //   analytics: 'granted' | 'denied',
  //   advertising: 'granted' | 'denied',
  //   timestamp: ISO string,
  //   version: 1,
  // }
  gtag('consent', 'update', {
    analytics_storage: e.detail.analytics,
    ad_storage: e.detail.advertising,
    ad_user_data: e.detail.advertising,
    ad_personalization: e.detail.advertising,
  });
  if (e.detail.advertising === 'granted') {
    fbq('consent', 'grant');
    // For California-detected visitors:
    // fbq('dataProcessingOptions', ['LDU'], 0, 0);
  } else {
    fbq('consent', 'revoke');
  }
});
